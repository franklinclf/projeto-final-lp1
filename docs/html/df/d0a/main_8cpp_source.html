52Colaborador *buscaBinaria(const set<Colaborador *, comparador> &equipe, int id)
53{
54 auto inicio = equipe.begin();
55 auto fim = equipe.end();
56
57 while (inicio != fim)
58 {
59 auto meio = std::next(inicio, std::distance(inicio, fim) / 2);
60
61 if ((*meio)->getID() == id)
62 {
63 return *meio;
64 }
65 else if ((*meio)->getID() < id)
66 {
67 inicio = std::next(meio);
68 }
69 else
70 {
71 fim = meio;
72 }
73 }
74
75 return nullptr;
76}
